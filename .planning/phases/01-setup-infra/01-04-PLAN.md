    ```python
    from functools import lru_cache
    from typing import Literal

    from langchain_openai import ChatOpenAI

    from app.config.settings import settings

    LLMRole = Literal["conversation", "judge", "intent_classifier", "summarizer"]

    # Aliases — both keys map to same model spec
    _ROLE_ALIASES: dict[str, LLMRole] = {
        "intent": "intent_classifier",
    }

    def _resolve_role(role: str) -> LLMRole:
        normalized = _ROLE_ALIASES.get(role, role)
        if normalized not in {"conversation", "judge", "intent_classifier", "summarizer"}:
            raise KeyError(f"Unknown LLM role: {role!r}")
        return normalized  # type: ignore[return-value]

    def _model_for(role: LLMRole) -> str:
        return {
            "conversation": settings.llm.model_conversation,
            "judge": settings.llm.model_judge,
            "intent_classifier": settings.llm.model_intent,
            "summarizer": settings.llm.model_summarizer,
        }[role]

    def _fallbacks_for(role: LLMRole) -> list[str]:
        return {
            "conversation": settings.llm.fallbacks_conversation,
            "judge": settings.llm.fallbacks_judge,
            "intent_classifier": [],
            "summarizer": [],
        }[role]

    def _temperature_for(role: LLMRole) -> float:
        return 0.0 if role == "judge" else 0.7

    @lru_cache(maxsize=8)
    def get_llm(role: str) -> ChatOpenAI:
        """Single point of LLM instantiation. Never instantiate ChatOpenAI directly."""
        resolved = _resolve_role(role)
        model = _model_for(resolved)
        fallbacks = _fallbacks_for(resolved)
        kwargs: dict = {
            "model": model,
            "base_url": settings.openrouter.base_url,
            "api_key": settings.openrouter.api_key.get_secret_value(),
            "default_headers": {
                "HTTP-Referer": settings.app.app_public_url,
                "X-Title": "landa-agent-service",
            },
            "temperature": _temperature_for(resolved),
            "timeout": 30,
            "max_retries": 2,
        }
        if fallbacks:
            kwargs["model_kwargs"] = {"models": fallbacks}
        return ChatOpenAI(**kwargs)

    # Public dict for visibility/debug; not used by get_llm directly
    ROLE_MODEL_MAP: dict[LLMRole, str] = {
        "conversation": settings.llm.model_conversation,
        "judge": settings.llm.model_judge,
        "intent_classifier": settings.llm.model_intent,
        "summarizer": settings.llm.model_summarizer,
    }
    ```

    Notas:
    - `@lru_cache(maxsize=8)` permite warm reuse del cliente HTTP interno de ChatOpenAI (sus connection pools persisten).
    - Headers `HTTP-Referer` + `X-Title` son los que OpenRouter usa para attribution/analytics (RESEARCH.md Pattern 2). NO incluyen PII.
    - Fallbacks vía `model_kwargs={"models": [...]}` es el patrón nativo de OpenRouter (RESEARCH.md Pattern 2 + Assumptions Log A8).
    - `temperature=0` para judge (determinístico — CLAUDE.md Stack tabla).
    - `timeout=30 + max_retries=2` defensivo; los retries con tenacity para el client HTTPx vienen en F2 si hace falta más.
    - LangSmith auto-tracing: con `LANGSMITH_TRACING=true` y `LANGSMITH_API_KEY` + `LANGSMITH_PROJECT` en env, `langchain` inserta tracing en cada ChatOpenAI sin código adicional. Es por esto que NO se pasa `callbacks=[]` ni `LangChainTracer()`.

    **Archivo 1: `app/healthcheck.py`** — router con `GET /health` (per RESEARCH.md Pattern 5):

    ```python
    import asyncio
    import time
    from typing import Any

    import httpx
    import structlog
    from fastapi import APIRouter, Request
    from sqlalchemy import text

    from app.config.settings import settings

    router = APIRouter(tags=["meta"])
    log = structlog.get_logger("healthcheck")


    async def _probe(coro, timeout: float = 1.0) -> dict[str, Any]:
        t0 = time.perf_counter()
        try:
            await asyncio.wait_for(coro, timeout=timeout)
            return {"ok": True, "latency_ms": round((time.perf_counter() - t0) * 1000, 1)}
        except Exception as e:  # noqa: BLE001 — health probe surfaces type only, not message
            return {
                "ok": False,
                "error": type(e).__name__,
                "latency_ms": round((time.perf_counter() - t0) * 1000, 1),
            }


    async def _check_postgres(request: Request) -> None:
        session_factory = request.app.state.session_factory
        async with session_factory() as s:
            await s.execute(text("SELECT 1"))


    async def _check_redis(request: Request) -> None:
        await request.app.state.redis.ping()


    async def _check_openrouter() -> None:
        # HEAD /api/v1 is cheap (no quota burn). Status 200/404 both prove DNS+TLS+reachability.
        async with httpx.AsyncClient(timeout=1.0) as c:
            r = await c.head(settings.openrouter.base_url)
            if r.status_code >= 500:
                raise RuntimeError(f"upstream {r.status_code}")


    @router.get("/health")
    async def health(request: Request) -> dict[str, Any]:
        pg, redis_, openrouter = await asyncio.gather(
            _probe(_check_postgres(request)),
            _probe(_check_redis(request)),
            _probe(_check_openrouter()),
        )
        langsmith_env = {
            "ok": bool(
                settings.langsmith.api_key
                and settings.langsmith.project
                and settings.langsmith.tracing
            )
        }
        components = {
            "postgres": pg,
            "redis": redis_,
            "openrouter": openrouter,
            "langsmith_env": langsmith_env,
        }
        status = "healthy" if all(c["ok"] for c in components.values()) else "degraded"
        # Always return HTTP 200 (RESEARCH.md Pattern 5 — Railway routes by 200, not by body)
        return {
            "status": status,
            "components": components,
            "version": settings.app.app_version,
            "env": settings.app.app_env,
        }
    ```

    **Archivo 2: `app/main.py`** — la app principal con lifespan + middleware + endpoints (per RESEARCH.md Pattern 1, 3, 4):

    ```python
    from contextlib import asynccontextmanager
    from collections.abc import AsyncIterator
    import time

    import structlog
    from asgi_correlation_id import CorrelationIdMiddleware
    from fastapi import FastAPI, Request
    from fastapi.responses import ORJSONResponse

    from app.config.checkpointer import build_checkpointer_cm
    from app.config.db import create_db_engine, create_session_factory
    from app.config.logging import configure_logging
    from app.config.observability import init_sentry
    from app.config.redis import close_redis_pool, create_redis_pool
    from app.config.settings import settings

    # IMPORTANT: configure logging + init Sentry BEFORE any router or feature import
    # (per RESEARCH.md Pattern 4 — Sentry auto-detection inspects sys.modules at init time)
    configure_logging(log_level=settings.app.log_level, env=settings.app.app_env)
    init_sentry()

    log = structlog.get_logger("main")

    from app.healthcheck import router as health_router  # noqa: E402 — after Sentry init
    from app.integrations.openrouter import get_llm       # noqa: E402


    @asynccontextmanager
    async def lifespan(app: FastAPI) -> AsyncIterator[None]:
        log.info("lifespan.startup", env=settings.app.app_env, version=settings.app.app_version)

        # 1. Postgres app engine (asyncpg)
        app.state.db_engine = create_db_engine()
        app.state.session_factory = create_session_factory(app.state.db_engine)

        # 2. Redis pool
        redis_client, redis_pool = create_redis_pool()
        app.state.redis = redis_client
        app.state.redis_pool = redis_pool

        # 3. LangGraph checkpointer (psycopg 3) — explicit __aenter__/__aexit__ pattern
        # (per CONTEXT.md D-06 + RESEARCH.md Pattern 1 / Pitfall 1)
        cp_cm = build_checkpointer_cm()
        app.state._cp_cm = cp_cm
        app.state.checkpointer = await cp_cm.__aenter__()
        # setup() is idempotent; safe to call even if alembic migration already ran
        await app.state.checkpointer.setup()

        log.info("lifespan.startup.complete")
        try:
            yield
        finally:
            log.info("lifespan.shutdown")
            await app.state._cp_cm.__aexit__(None, None, None)
            await close_redis_pool(app.state.redis, app.state.redis_pool)
            await app.state.db_engine.dispose()
            log.info("lifespan.shutdown.complete")


    app = FastAPI(
        title="landa-agent-service",
        version=settings.app.app_version,
        default_response_class=ORJSONResponse,
        lifespan=lifespan,
    )

    # asgi-correlation-id MUST be added before any other middleware that wants the ID
    app.add_middleware(CorrelationIdMiddleware, header_name="X-Request-ID")


    @app.middleware("http")
    async def bind_correlation_to_structlog(request: Request, call_next):
        # asgi-correlation-id sets correlation_id in contextvars; structlog merge_contextvars
        # picks it up automatically. This middleware also logs request boundaries.
        from asgi_correlation_id.context import correlation_id

        structlog.contextvars.bind_contextvars(
            correlation_id=correlation_id.get() or "-",
            path=request.url.path,
            method=request.method,
        )
        t0 = time.perf_counter()
        try:
            response = await call_next(request)
        finally:
            log.info(
                "request.complete",
                status=getattr(locals().get("response", None), "status_code", 500),
                latency_ms=round((time.perf_counter() - t0) * 1000, 1),
            )
            structlog.contextvars.clear_contextvars()
        return response


    app.include_router(health_router)


    @app.post("/test/llm")
    async def test_llm(payload: dict | None = None) -> dict:
        """Dummy endpoint that verifies the OpenRouter + LangSmith pipeline end-to-end.

        ROADMAP F1 deliverable 12. NOT for production use — remove or auth-gate in F5.

        """

    ```
